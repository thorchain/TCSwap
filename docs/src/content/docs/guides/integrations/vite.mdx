## Vite Configuration

Update your `vite.config.ts` to include necessary polyfills and WebAssembly support:

```typescript
// @noErrorValidation

import { defineConfig } from "vite";
import react from "@vitejs/plugin-react";
import wasm from "vite-plugin-wasm";
import topLevelAwait from "vite-plugin-top-level-await";
import { nodePolyfills } from "vite-plugin-node-polyfills";

export default defineConfig({
  plugins: [
    react(),
    wasm(),
    topLevelAwait(),
    nodePolyfills({
      exclude: ["fs"],
      globals: {
        Buffer: true,
        global: true,
        process: true,
      },
      protocolImports: true,
    }),
  ],
  define: {
    "process.env": {},
    global: "globalThis",
  },
  optimizeDeps: {
    esbuildOptions: {
      target: "esnext",
    },
  },
  build: {
    target: "esnext",
  },
});
```

## Basic Setup

Create a USwap client instance:

```typescript
// @noErrorValidation

import { createUSwap } from "@tcswap/sdk";

export const uSwapClient = createUSwap();
```

## React Component Example

Here's a basic example of using USwap in a React component:

```tsx
import { useState, useEffect } from "react";
import { uSwapClient } from "./uSwapClient";
import { Chain, WalletOption } from "@tcswap/sdk";

function App() {
  const [isConnected, setIsConnected] = useState(false);
  const [address, setAddress] = useState<string>("");

  const connectWallet = async () => {
    try {
      await uSwapClient.connectEVMWallet(
        [Chain.Ethereum],
        WalletOption.METAMASK
      );

      setIsConnected(true);
      const ethAddress = await uSwapClient.getAddress(Chain.Ethereum);
      setAddress(ethAddress);
    } catch (error) {
      console.error("Failed to connect wallet:", error);
    }
  };

  const disconnectWallet = () => {
    uSwapClient.disconnectWallet(Chain.Ethereum);
    setIsConnected(false);
    setAddress("");
  };

  return (
    <div>
      <h1>USwap + Vite</h1>
      {!isConnected ? (
        <button onClick={connectWallet}>Connect Wallet</button>
      ) : (
        <div>
          <p>Connected: {address}</p>
          <button onClick={disconnectWallet}>Disconnect</button>
        </div>
      )}
    </div>
  );
}

export default App;
```

## Environment Variables

For environment variables in Vite, use the `VITE_` prefix:

```bash
# .env
VITE_USWAP_API_KEY=your_api_key_here
VITE_WALLETCONNECT_PROJECT_ID=your_project_id_here
```

Then configure USwap:

```typescript
// @noErrorValidation

import { createUSwap, USwapConfig } from "@tcswap/sdk";

USwapConfig.setApiKey("uSwap", import.meta.env.VITE_USWAP_API_KEY);
USwapConfig.setApiKey(
  "walletConnectProjectId",
  import.meta.env.VITE_WALLETCONNECT_PROJECT_ID
);

export const uSwapClient = createUSwap();
```
