## Two Approaches to Using USwap

You can use USwap in two ways:

1. **Granular approach** - Import only what you need from individual packages:

   ```ts
   import { USwap, keystoreWallet } from "@tcswap/sdk";
   ```

2. **All-in-one approach** - Import everything from the SDK (better for backend/Node.js):
   ```ts
   import { createUSwap } from "@tcswap/sdk";
   ```

The granular approach is recommended for frontend applications as it results in smaller bundle sizes.

## Initialize USwap

### Granular Approach (Frontend Recommended)

```ts twoslash
import { USwap, keystoreWallet, ledgerWallet, evmWallet } from "@tcswap/sdk";

const uSwap = USwap({
  config: {
    apiKeys: {
      uSwap: "your-uswap-api-key",
      walletConnectProjectId: "your-walletconnect-project-id",
    },
  },
  wallets: { ...keystoreWallet, ...ledgerWallet, ...evmWallet },
});
```

### All-in-one Approach (Backend/Node.js)

```ts twoslash
import { createUSwap } from "@tcswap/sdk";

const uSwap = createUSwap({
  config: {
    apiKeys: {
      uSwap: "your-uswap-api-key",
      walletConnectProjectId: "your-walletconnect-project-id",
    },
  },
});
```

## Connect with Keystore Wallet

### Granular Approach

```ts twoslash
import {
  Chain,
  USwap,
  keystoreWallet,
  decryptFromKeystore,
  type Keystore,
} from "@tcswap/sdk";

const uSwap = USwap({
  wallets: { ...keystoreWallet },
});

async function connectWithKeystore(
  keystoreFile: Keystore,
  password: string,
  chains: Chain[]
) {
  try {
    const phrase = await decryptFromKeystore(keystoreFile, password);

    await uSwap.connectKeystore(chains, phrase);

    const balances = await Promise.all(
      chains.map(async (chain) => {
        const wallet = await uSwap.getWalletWithBalance(chain);
        return wallet.balance;
      })
    );

    return { success: true, balances };
  } catch (error) {
    return { success: false, error };
  }
}

const chains = [Chain.Ethereum, Chain.Bitcoin, Chain.Ripple];
await connectWithKeystore(keystoreFile, password, chains);
```

### All-in-one Approach

```ts twoslash
import {
  Chain,
  createUSwap,
  decryptFromKeystore,
  type Keystore,
} from "@tcswap/sdk";

const uSwap = createUSwap();

async function connectWithKeystore(
  keystoreFile: Keystore,
  password: string,
  chains: Chain[]
) {
  try {
    const phrase = await decryptFromKeystore(keystoreFile, password);

    await uSwap.connectKeystore(chains, phrase);

    return { success: true };
  } catch (error) {
    return { success: false, error };
  }
}
```

## Using Hardware Wallets

### Ledger Wallet

```ts twoslash
import { Chain, USwap, ledgerWallet } from "@tcswap/sdk";

const uSwap = USwap({
  wallets: { ...ledgerWallet },
});

async function connectLedger(chains: Chain[]) {
  try {
    await uSwap.connectLedger(chains);

    const ethAddress = uSwap.getAddress(Chain.Ethereum);

    return { success: true };
  } catch (error) {
    return { success: false, error };
  }
}
```

### Trezor Wallet

```ts twoslash
import {
  Chain,
  NetworkDerivationPath,
  USwap,
  trezorWallet,
} from "@tcswap/sdk";

const uSwap = USwap({
  wallets: { ...trezorWallet },
});

async function connectTrezor(chains: Chain[]) {
  try {
    await uSwap.connectTrezor([Chain.Bitcoin], NetworkDerivationPath.BTC);

    const btcWallet = await uSwap.getWalletWithBalance(Chain.Bitcoin);

    return { success: true, balance: btcWallet.balance };
  } catch (error) {
    return { success: false, error };
  }
}
```

## Managing Wallet Connections

```ts twoslash
import { Chain, USwap, evmWallet } from "@tcswap/sdk";

const uSwap = USwap({
  wallets: { ...evmWallet },
});

function getAllWallets() {
  const wallets = uSwap.getAllWallets();
  return wallets;
}

function disconnectChain(chain: Chain) {
  uSwap.disconnectChain(chain);
}

function disconnectAll() {
  uSwap.disconnectAll();
}
```

## Getting Wallet Balances

```ts twoslash
import { Chain, USwap, evmWallet } from "@tcswap/sdk";

const uSwap = USwap({
  wallets: { ...evmWallet },
});

await uSwap.connectEVMWallet([Chain.Ethereum]);

async function getChainBalance(chain: Chain) {
  const balance = await uSwap.getBalance(chain, true);
  return balance;
}

async function getAllBalances(chains: Chain[]) {
  const balances = await Promise.all(
    chains.map(async (chain) => {
      try {
        const wallet = await uSwap.getWalletWithBalance(chain);
        return { chain, balance: wallet.balance };
      } catch (error) {
        return { chain, balance: [], error };
      }
    })
  );

  return balances;
}
```

## Using WalletConnect

```ts twoslash

const uSwap = USwap({
  config: {
    apiKeys: {
      walletConnectProjectId: "YOUR_WALLETCONNECT_PROJECT_ID",
    },
  },
  wallets: { ...walletconnectWallet },
});

async function connectWalletConnect() {
  try {
    await uSwap.connectWalletconnect([
      Chain.Ethereum,
      Chain.BinanceSmartChain,
      Chain.Cosmos,
    ]);

    const ethAddress = uSwap.getAddress(Chain.Ethereum);

    return { success: true };
  } catch (error) {
    return { success: false, error };
  }
}
```

## Advanced: Connecting Multiple Wallets

```ts twoslash
import {
  Chain,
  USwap,
  evmWallet,
  keplrWallet,
  keystoreWallet,
} from "@tcswap/sdk";

const uSwap = USwap({
  wallets: { ...evmWallet, ...keplrWallet, ...keystoreWallet },
});

async function connectMultipleWallets(phrase: string) {
  try {
    await uSwap.connectEVMWallet([Chain.Ethereum]);

    await uSwap.connectKeplr([Chain.Cosmos]);

    await uSwap.connectKeystore([Chain.Bitcoin], phrase);

    const wallets = uSwap.getAllWallets();

    const addresses = {
      [Chain.Ethereum]: uSwap.getAddress(Chain.Ethereum),
      [Chain.Cosmos]: uSwap.getAddress(Chain.Cosmos),
      [Chain.Bitcoin]: uSwap.getAddress(Chain.Bitcoin),
    };

    return { success: true, addresses };
  } catch (error) {
    return { success: false, error };
  }
}
```
